    function displayLink(id, url, text) {
        // Create link

        var linkBar = document.createElement('div');

    * Removes an element from the DOM

    function removeLink(id) {

        // Check browser compatibility
        isIE = (navigator.appVersion.indexOf('MSIE') != -1) ? true : false;
        if (isIE) {

    function increateRight(id, max) {
        // Get element
        var linkBar = document.getElementById(id);
        if (linkBar) {
            // Get the CSS 'right' position
            var right = 0;
